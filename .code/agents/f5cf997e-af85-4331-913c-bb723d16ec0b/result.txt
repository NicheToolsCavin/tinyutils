#!/usr/bin/env python3
"""Vendor pandoc 3.1.11.1 (linux-amd64) into tinyutils/api/_vendor/pandoc/pandoc."""

from __future__ import annotations

import hashlib
import os
import shutil
import tarfile
import tempfile
from pathlib import Path
from urllib import request, error

PANDOC_VERSION = "3.1.11.1"
ARCHIVE_NAME = f"pandoc-{PANDOC_VERSION}-linux-amd64.tar.gz"
DOWNLOAD_URL = f"https://github.com/jgm/pandoc/releases/download/{PANDOC_VERSION}/{ARCHIVE_NAME}"
CHUNK_SIZE = 1024 * 1024
TARGET_DIR = Path(__file__).resolve().parents[1] / "_vendor" / "pandoc"
TARGET_BIN = TARGET_DIR / "pandoc"


def sha256sum(path: Path) -> str:
    digest = hashlib.sha256()
    with path.open("rb") as fh:
        for chunk in iter(lambda: fh.read(CHUNK_SIZE), b""):
            digest.update(chunk)
    return digest.hexdigest()


def fetch(url: str, destination: Path) -> None:
    try:
        with request.urlopen(url) as response, destination.open("wb") as dst:
            shutil.copyfileobj(response, dst, CHUNK_SIZE)
    except error.URLError as exc:
        raise SystemExit(f"Download failed: {exc}") from exc


def extract_pandoc(archive: Path, destination: Path) -> None:
    with tarfile.open(archive, "r:gz") as bundle:
        member = next(
            (m for m in bundle.getmembers() if m.isfile() and m.name.endswith("/pandoc")),
            None,
        )
        if member is None:
            raise SystemExit("pandoc binary not found in archive")
        destination.parent.mkdir(parents=True, exist_ok=True)
        with bundle.extractfile(member) as src, destination.open("wb") as dst:
            shutil.copyfileobj(src, dst, CHUNK_SIZE)


def install() -> None:
    TARGET_DIR.mkdir(parents=True, exist_ok=True)
    with tempfile.TemporaryDirectory(prefix="pandoc-vendor-") as tmp_dir:
        tmp_dir = Path(tmp_dir)
        archive_path = tmp_dir / ARCHIVE_NAME
        print(f"Fetching {DOWNLOAD_URL} …")
        fetch(DOWNLOAD_URL, archive_path)
        archive_sha = sha256sum(archive_path)

        extracted = tmp_dir / "pandoc-bin"
        extract_pandoc(archive_path, extracted)

        if TARGET_BIN.exists():
            TARGET_BIN.unlink()
        shutil.move(str(extracted), TARGET_BIN)

    os.chmod(TARGET_BIN, 0o755)
    binary_sha = sha256sum(TARGET_BIN)
    print(f"pandoc {PANDOC_VERSION} installed at {TARGET_BIN}")
    print("Checksums:")
    print(f"  archive sha256: {archive_sha}")
    print(f"  binary  sha256: {binary_sha}")


if __name__ == "__main__":
    install()
```
- Downloads the requested tarball, extracts only the binary, enforces `0755`, and prints archive/binary SHA256 summaries.
- Uses only stdlib so it’s portable and Vercel-friendly; no downloads run until someone executes the script.

from pathlib import Path
import os
import shutil
import subprocess

PANDOC_ENV_VAR = "PYPANDOC_PANDOC"
VENDORED_PANDOC = Path(__file__).resolve().parents[1] / "_vendor" / "pandoc" / "pandoc"


def resolve_pandoc_binary() -> str:
    env_path = os.getenv(PANDOC_ENV_VAR)
    if env_path:
        return env_path

    if VENDORED_PANDOC.exists():
        return str(VENDORED_PANDOC)

    system_path = shutil.which("pandoc")
    if system_path:
        return system_path

    raise RuntimeError(
        "pandoc is not available; set PYPANDOC_PANDOC, install pandoc, "
        "or run api/_scripts/vendor_pandoc.py."
    )


def run_pandoc(args: list[str], stdin: str | None = None) -> subprocess.CompletedProcess[str]:
    binary = resolve_pandoc_binary()
    return subprocess.run(
        [binary, *args],
        check=True,
        input=stdin,
        text=True,
        capture_output=True,
    )
```
- Prefers `PYPANDOC_PANDOC`, falls back to vendored binary, then system `pandoc`, meeting the acceptance criteria.
