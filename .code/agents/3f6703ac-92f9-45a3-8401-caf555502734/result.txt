## 5. Alert Thresholds and Validation Steps

### Alert Thresholds

#### High Priority (Immediate Action Required):
- **5xx Error Rate > 5% for 5+ consecutive minutes**
  - Slack/Discord notification to #alerts channel
- **Average Response Time > 5 seconds for 2+ consecutive minutes**
  - Page on-call engineer
- **Cache Hit Rate < 10% for 10+ consecutive minutes**
  - Indicates potential cache malfunction
- **Blob Upload Failure Rate > 20% for 5+ consecutive minutes**
  - Could indicate Vercel Blob API issues or token problems
- **API Completely Unresponsive** (health checks failing)

#### Medium Priority (Review Within 4 Hours):
- **4xx Error Rate > 20% for 5+ consecutive minutes**
  - Client-side issues, but worth investigating
- **Average Response Time > 2 seconds for 5+ consecutive minutes**
  - Performance degradation
- **Pandoc unavailable for > 1 minute**
  - Conversion functionality will be impaired
- **Cache Hit Rate drops by > 50% from baseline**
  - Could indicate cache clearing or invalidation issues

#### Low Priority (Review Within 24 Hours):
- **Request rate drops > 50% from baseline**
  - Could indicate service discovery or routing issues
- **Cache hit rate < 50% but > 10%**
  - Normal operation, but worth monitoring

def upload_bytes(name: str, data: bytes, content_type: str) -> str:
    """Upload bytes to blob storage or fall back to a data URL."""
    start_time = time.time()

    token = os.getenv("BLOB_READ_WRITE_TOKEN")
    if token:
        try:
            blob_url = _upload_to_vercel_blob(name, data, content_type, token)
            if blob_url:
                duration_ms = (time.time() - start_time) * 1000
                logger.info(
                    "blob upload successful name=%s size=%d duration_ms=%.2f",
                    name,
                    len(data),
                    duration_ms
                )
                return blob_url
        except requests.RequestException as exc:
            duration_ms = (time.time() - start_time) * 1000
            logger.error(
                "blob upload failed name=%s size=%d duration_ms=%.2f error=%s",
                name,
                len(data),
                duration_ms,
                str(exc)
            )
        except ValueError as exc:
            duration_ms = (time.time() - start_time) * 1000
            logger.error(
                "blob upload invalid response name=%s size=%d duration_ms=%.2f error=%s",
                name,
                len(data),
                duration_ms,
                str(exc)
            )

    # Fallback to data URL
    duration_ms = (time.time() - start_time) * 1000
    logger.warning(
        "blob upload fallback to data url name=%s size=%d duration_ms=%.2f",
        name,
        len(data),
        duration_ms
    )
    encoded = base64.b64encode(data).decode("ascii")
    return f"data:{content_type};base64,{encoded}"


def _upload_to_vercel_blob(
    name: str,
    data: bytes,
    content_type: str,
    token: str,
) -> Optional[str]:
    start_time = time.time()
    headers = {
        "Authorization": f"Bearer {token}",
        "User-Agent": USER_AGENT,
    }
    files = {"file": (name, data, content_type)}
    
    response = requests.post(
        BLOB_UPLOAD_URL,
        headers=headers,
        files=files,
        timeout=DEFAULT_TIMEOUT,
    )
    
    duration_ms = (time.time() - start_time) * 1000
    
    try:
        response.raise_for_status()
        logger.info(
            "blob api request successful name=%s size=%d duration_ms=%.2f status=%d",
            name,
            len(data),
            duration_ms,
            response.status_code
        )
    except requests.HTTPError:
        logger.error(
            "blob api request failed name=%s size=%d duration_ms=%.2f status=%d error=%s",
            name,
            len(data),
            duration_ms,
            response.status_code,
            response.text
        )
        raise
    
    try:
        payload = response.json()
    except json.JSONDecodeError as exc:
        logger.error(
            "blob api invalid json response name=%s size=%d duration_ms=%.2f error=%s",
            name,
            len(data),
            duration_ms,
            str(exc)
        )
        raise ValueError("Blob API returned non-JSON payload") from exc

    blob_url = (
        payload.get("url")
        or payload.get("blob", {}).get("url")
        or payload.get("pathname")
    )
    
    if not blob_url:
        logger.error(
            "blob api missing url in response name=%s size=%d duration_ms=%.2f payload=%s",
            name,
            len(data),
            duration_ms,
            payload
        )
    
    return blob_url
```

"""Health check endpoint for the convert API."""
from __future__ import annotations

import logging
import time
from typing import Dict, Any

from fastapi import FastAPI, HTTPException

from tinyutils.api._lib.pandoc_runner import get_pandoc_version, ensure_pandoc
from tinyutils.api._lib import blob

logger = logging.getLogger(__name__)

app = FastAPI()

@app.get("/health")
def health_check() -> Dict[str, Any]:
    """Health check for the convert API and its dependencies."""
    checks = {
        "pandoc": False,
        "blob_storage": False
    }
    
    start_time = time.time()
    
    try:
        # Check pandoc availability
        version = get_pandoc_version()
        if version:
            checks["pandoc"] = True
    except Exception as e:
        logger.error(f"Health check failed for pandoc: {e}")
    
    try:
        # Check blob storage (try a small upload)
        test_data = b"healthcheck"
        test_url = blob.upload_bytes("health-check.txt", test_data, "text/plain")
        checks["blob_storage"] = test_url.startswith(("https://", "data:"))
    except Exception as e:
        logger.error(f"Health check failed for blob storage: {e}")
    
    overall_health = all(checks.values())
    
    response = {
        "status": "healthy" if overall_health else "unhealthy",
        "checks": checks,
        "timestamp": time.time(),
        "response_time_ms": (time.time() - start_time) * 1000
    }
    
    if not overall_health:
        raise HTTPException(status_code=503, detail=response)
    
    return response
```
