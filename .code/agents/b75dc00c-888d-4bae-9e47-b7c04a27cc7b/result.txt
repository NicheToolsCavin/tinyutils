// Add artifact persistence function
async function persistArtifact(request, response, testName) {
  if (process.env.ARTIFACT_DIR) {
    const fs = require('fs').promises;
    const path = require('path');
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    const fileName = `${testName}_${timestamp}.json`;
    const filePath = path.join(ARTIFACT_DIR, fileName);
    
    const artifact = {
      request: maskSensitiveData(request),
      response: {
        statusCode: response.status,
        headers: response.headers.raw(),
        body: await response.text()
      },
      timestamp: new Date().toISOString()
    };
    
    await fs.mkdir(ARTIFACT_DIR, { recursive: true });
    await fs.writeFile(filePath, JSON.stringify(artifact, null, 2));
    console.log(`Artifact saved: ${filePath}`);
  }
}

// Security masking helper
function maskSensitiveData(data) {
  // Mask common secret patterns
  const sensitivePatterns = ['authorization', 'cookie', 'x-api-key', 'x-auth-token'];
  if (typeof data === 'object') {
    const masked = { ...data };
    for (const [key, value] of Object.entries(masked)) {
      if (sensitivePatterns.some(pattern => key.toLowerCase().includes(pattern))) {
        masked[key] = '[MASKED]';
      }
    }
    return masked;
  }
  return data;
}

## Security Masking Rules
- Mask headers containing: `authorization`, `x-api-key`, `x-auth-token`, `x-csrf-token`
- Mask cookie values containing: `sessionid`, `auth_token`, `access_token`, `refresh_token`
- Mask any header with "key", "token", "secret", "auth" in the name (case-insensitive)
- Redact values for sensitive headers in logs and artifacts
