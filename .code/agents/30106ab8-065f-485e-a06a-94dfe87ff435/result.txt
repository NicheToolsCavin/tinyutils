  ```bash
  #!/usr/bin/env bash
  # shellcheck shell=bash

  set -euo pipefail

  readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
  readonly PROJECT_ROOT="$(cd "${SCRIPT_DIR}/.." && pwd)"
  readonly MADRID_TZ="Europe/Madrid"

  readonly UTC_STAMP="$(date -u +"%Y%m%dT%H%M%SZ")"
  readonly MADRID_STAMP="$(TZ="${MADRID_TZ}" date +"%Y%m%dT%H%M%S")"

  readonly ARTIFACT_DIR="${PROJECT_ROOT}/artifacts/convert/${UTC_STAMP}/cred-check-${MADRID_STAMP}"
  readonly LOG_FILE="${ARTIFACT_DIR}/vercel_whoami.log"
  readonly PROJECT_CONFIG="${PROJECT_ROOT}/.vercel/project.json"

  mkdir -p "${ARTIFACT_DIR}"
  : > "${LOG_FILE}"

  log_note() {
    printf '%s\n' "$1" >>"${LOG_FILE}"
  }

  log_warn() {
    printf 'WARN: %s\n' "$1" >>"${LOG_FILE}"
  }

  run_vercel() {
    if command -v vercel >/dev/null 2>&1; then
      vercel "$@"
      return
    fi

    local -r local_bin="${PROJECT_ROOT}/node_modules/.bin/vercel"
    if [[ -x "${local_bin}" ]]; then
      "${local_bin}" "$@"
      return
    fi

    if command -v pnpm >/dev/null 2>&1; then
      pnpm dlx vercel@latest "$@"
      return
    fi

    if command -v npx >/dev/null 2>&1; then
      npx --yes vercel@latest "$@"
      return
    fi

    printf 'vercel CLI not found; install pnpm or npm to proceed.\n' >&2
    exit 1
  }

  declare -a COMMON_ARGS=()
  if [[ -n "${VERCEL_TOKEN:-}" ]]; then
    COMMON_ARGS+=("--token" "${VERCEL_TOKEN}")
  fi
  if [[ -n "${VERCEL_SCOPE:-}" ]]; then
    COMMON_ARGS+=("--scope" "${VERCEL_SCOPE}")
  elif [[ -n "${VERCEL_ORG_ID:-}" ]]; then
    COMMON_ARGS+=("--scope" "${VERCEL_ORG_ID}")
  fi

  log_note "UTC run: ${UTC_STAMP}"
  log_note "Madrid run: ${MADRID_STAMP}"
  log_note "== vercel whoami =="

  if ! run_vercel whoami "${COMMON_ARGS[@]}" >>"${LOG_FILE}" 2>&1; then
    printf 'Vercel authentication failed; see %s\n' "${LOG_FILE}" >&2
    exit 1
  fi

  log_note "== vercel whoami complete =="

  if [[ ! -f "${PROJECT_CONFIG}" ]]; then
    if [[ -n "${VERCEL_PROJECT_ID:-}" && -n "${VERCEL_ORG_ID:-}" ]]; then
      log_note "Project not linked; attempting non-interactive vercel link."
      if ! run_vercel link --yes --project "${VERCEL_PROJECT_ID}" "${COMMON_ARGS[@]}" >>"${LOG_FILE}" 2>&1; then
        printf 'Vercel link failed; see %s\n' "${LOG_FILE}" >&2
        exit 1
      fi
    else
      log_warn "Project not linked and VERCEL_PROJECT_ID/VERCEL_ORG_ID missing; skipping vercel link."
    fi
  fi

  log_note "Credential check completed."
  printf 'vercel credentials OK -> %s\n' "${LOG_FILE}"
  ```
