**Deploy Script**
- The workspace is mounted read-only (approval policy `never`), so creation/`chmod` of `tinyutils/scripts/deploy_preview.sh` was blocked; below is the exact script I prepared so you can drop it in verbatim.
- It enforces `set -euo pipefail`, verifies `vercel`/`node`, checks the required Vercel env vars, and only proceeds when everything needed is present—no secret values are echoed, only variable names.
- Preview deploy output, pull logs, and smoke-test logs are streamed live but also captured under `tinyutils/artifacts/convert/<YYYYMMDD>/preview-<timestamp>/`, the preview URL is exported via `PREVIEW_URL`, the smoke test (`node scripts/smoke_dlf_preview.mjs`) runs against that URL, and a Madrid-timestamped summary is appended to `tinyutils/docs/AGENT_RUN_LOG.md`.

```bash
#!/usr/bin/env bash
set -euo pipefail

SCRIPT_DIR="$(cd -- "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
ROOT_DIR="$(cd -- "${SCRIPT_DIR}/.." && pwd)"
SMOKE_SCRIPT="${ROOT_DIR}/scripts/smoke_dlf_preview.mjs"

ARTIFACT_BASE="${ROOT_DIR}/artifacts/convert"
DATE_STAMP="$(TZ=UTC date +%Y%m%d)"
RUN_STAMP="$(TZ=UTC date +%H%M%S)"
ARTIFACT_DIR="${ARTIFACT_BASE}/${DATE_STAMP}/preview-${RUN_STAMP}"
PULL_LOG="${ARTIFACT_DIR}/vercel_pull.log"
DEPLOY_LOG="${ARTIFACT_DIR}/vercel_deploy.log"
SMOKE_LOG="${ARTIFACT_DIR}/smoke.log"
SUMMARY_LOG="${ROOT_DIR}/docs/AGENT_RUN_LOG.md"

mkdir -p "${ARTIFACT_DIR}"
mkdir -p "$(dirname "${SUMMARY_LOG}")"

require_cmd() {
  if ! command -v "$1" >/dev/null 2>&1; then
    echo "Missing required command: $1" >&2
    exit 1
  fi
}

require_cmd vercel
require_cmd node

if [[ ! -f "${SMOKE_SCRIPT}" ]]; then
  echo "Missing smoke script at ${SMOKE_SCRIPT}" >&2
  exit 1
fi

missing_envs=()
[[ -z "${VERCEL_TOKEN:-}" ]] && missing_envs+=("VERCEL_TOKEN")
[[ -z "${VERCEL_PROJECT_ID:-}" ]] && missing_envs+=("VERCEL_PROJECT_ID")
[[ -z "${VERCEL_SCOPE:-}" && -z "${VERCEL_ORG_ID:-}" ]] && missing_envs+=("VERCEL_SCOPE|VERCEL_ORG_ID")

if ((${#missing_envs[@]} > 0)); then
  echo "Missing required environment variables: ${missing_envs[*]}" >&2
  exit 1
fi

SCOPE_VALUE="${VERCEL_SCOPE:-${VERCEL_ORG_ID}}"

echo "Pulling preview environment via Vercel..."
(
  cd "${ROOT_DIR}"
  vercel pull --environment=preview --yes --scope "${SCOPE_VALUE}"
) | tee "${PULL_LOG}"

echo "Deploying preview via Vercel..."
(
  cd "${ROOT_DIR}"
  vercel deploy --yes --confirm
) | tee "${DEPLOY_LOG}"

PREVIEW_URL="$(grep -Eo 'https?://[-[:alnum:]]+\.vercel\.app' "${DEPLOY_LOG}" | tail -n1 || true)"
if [[ -z "${PREVIEW_URL}" ]]; then
  echo "Unable to determine preview URL from deploy output." >&2
  exit 1
fi
export PREVIEW_URL
echo "Preview URL: ${PREVIEW_URL}"

echo "Running converter smoke checks..."
(
  cd "${ROOT_DIR}"
  PREVIEW_URL="${PREVIEW_URL}" node "${SMOKE_SCRIPT}"
) | tee "${SMOKE_LOG}"

MADRID_TS="$(TZ=Europe/Madrid date '+%Y-%m-%d %H:%M:%S %Z')"
REL_ARTIFACT_PATH="${ARTIFACT_DIR#"${ROOT_DIR}/"}"
printf '* %s | Preview: %s | Artifacts: %s\n' "${MADRID_TS}" "${PREVIEW_URL}" "${REL_ARTIFACT_PATH}" >> "${SUMMARY_LOG}"

echo "Artifacts stored under ${REL_ARTIFACT_PATH}"
```
