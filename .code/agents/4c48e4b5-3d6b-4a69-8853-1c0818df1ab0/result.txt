+function sliceFunction(source, signature) {
+  const start = source.indexOf(signature);
+  if (start === -1) {
+    throw new Error(`Unable to locate ${signature.trim()} in ${helpersPath}`);
+  }
+  const searchStart = start + signature.length;
+  const candidates = [source.indexOf('\nfunction ', searchStart), source.indexOf('\n    function', searchStart)]
+    .filter((value) => value !== -1);
+  const end = candidates.length ? Math.min(...candidates) : source.length;
+  return source.slice(start, end).trim();
+}
