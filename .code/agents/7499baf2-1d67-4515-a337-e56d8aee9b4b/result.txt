+def _compute_cache_key(
+    *,
+    input_bytes: bytes,
+    targets: Sequence[str],
+    options: ConversionOptions,
+    from_format: Optional[str],
+) -> str:
+    digest = hashlib.sha256()
+    digest.update(CACHE_VERSION.encode("utf-8"))
+    digest.update(b"\x00")
+    digest.update(",".join(targets).encode("utf-8"))
+    digest.update(b"\x00")
+    digest.update(b"1" if options.accept_tracked_changes else b"0")
+    digest.update(b"1" if options.extract_media else b"0")
+    digest.update(b"1" if options.remove_zero_width else b"0")
+    digest.update(b"\x00")
+    digest.update((from_format or "").strip().lower().encode("utf-8"))
+    digest.update(b"\x00")
+    digest.update(len(input_bytes).to_bytes(8, "big"))
+    digest.update(input_bytes)
+    return digest.hexdigest()
